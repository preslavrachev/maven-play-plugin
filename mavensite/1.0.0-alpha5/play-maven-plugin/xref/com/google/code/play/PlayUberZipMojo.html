126     protected void internalExecute()
127         throws MojoExecutionException, MojoFailureException, IOException
128     {
129         if (uberzipSkip)
130         {
131             getLog().info( "UberZip generation skipped" );
132             return;
133         }
134         
135         try
136         {
137             File baseDir = project.getBasedir();
138             File destFile = new File( uberzipOutputDirectory, getDestinationFileName() );
139 
140             Archiver zipArchiver = archiverManager.getArchiver( "zip" );
141             zipArchiver.setDuplicateBehavior( Archiver.DUPLICATES_FAIL );// Just in case
142             zipArchiver.setDestFile( destFile );
143 
144             // APPLICATION
145             getLog().debug( "UberZip includes: " + uberzipIncludes );
146             getLog().debug( "UberZip excludes: " + uberzipExcludes );
147             String[] includes = ( uberzipIncludes != null ? uberzipIncludes.split( "," ) : null );
148             String[] excludes = ( uberzipExcludes != null ? uberzipExcludes.split( "," ) : null );
149             zipArchiver.addDirectory( baseDir, "application/", includes, excludes );
150 
151             //framework
152             Artifact frameworkArtifact = findFrameworkArtifact( false );
153             File frameworkZipFile = frameworkArtifact.getFile();
154             zipArchiver.addArchivedFileSet( frameworkZipFile );
155 
156             //modules
157             Map<String, Artifact> moduleArtifacts = findAllModuleArtifacts( false );
158             for ( Map.Entry<String, Artifact> moduleArtifactEntry : moduleArtifacts.entrySet() )
159             {
160                 String moduleName = moduleArtifactEntry.getKey();
161                 Artifact moduleArtifact = moduleArtifactEntry.getValue();
162 
163                 File moduleZipFile = moduleArtifact.getFile();
164                 String moduleSubDir =
165                     String.format( "application/modules/%s-%s/", moduleName, moduleArtifact.getVersion() );
166                 if ( Artifact.SCOPE_PROVIDED.equals( moduleArtifact.getScope() ) )
167                 {
168                     moduleSubDir = String.format( "modules/%s/", moduleName/* , moduleArtifact.getVersion() */);
169                 }
170                 zipArchiver.addArchivedFileSet( moduleZipFile, moduleSubDir );
171             }
172 
173             Set<?> artifacts = project.getArtifacts();
174             for ( Iterator<?> iter = artifacts.iterator(); iter.hasNext(); )
175             {
176                 Artifact artifact = (Artifact) iter.next();
177                 if ( "jar".equals( artifact.getType() ) )
178                 {
179                     File jarFile = artifact.getFile();
180                     String destinationFileName = "application/lib/" + jarFile.getName();
181                     // Play! Framework's library
182                     if ( Artifact.SCOPE_PROVIDED.equals( artifact.getScope() ) )
183                     {
184                         // check if there is module zip dependency
185                         String moduleName = artifact.getArtifactId();
186                         if ( artifact.getArtifactId().startsWith( "play-" ) )
187                         {
188                             moduleName = moduleName.substring( "play-".length() );
189                         }
190                         if ( moduleArtifacts.containsKey( moduleName ) )
191                         {
192                             destinationFileName = String.format( "modules/%s/lib/%s", moduleName, jarFile.getName() );
193                         }
194                         else
195                         {
196                             destinationFileName = "framework/lib/" + jarFile.getName();
197                             if ( "play".equals( artifact.getArtifactId() ) ) // ???
198                             {
199                                 destinationFileName = "framework/" + jarFile.getName();
200                                 String playVersion = artifact.getVersion();
201                                 if ( "1.2".compareTo( playVersion ) > 0 )
202                                 {
203                                     // Play 1.1.x
204                                     destinationFileName = "framework/play.jar";
205                                 }
206                             }
207                         }
208                     }
209                     zipArchiver.addFile( jarFile, destinationFileName );
210                 }
211             }
212 
213             zipArchiver.createArchive();
214         }
215         catch ( ArchiverException e )
216         {
217             throw new MojoExecutionException( "?", e );
218         }
219         catch ( NoSuchArchiverException e )
220         {
221             throw new MojoExecutionException( "?", e );
222         }
223     }
224 
225     private String getDestinationFileName()
226     {
227         StringBuffer buf = new StringBuffer();
228         buf.append( uberzipArchiveName );
229         if ( uberzipClassifier != null && !"".equals( uberzipClassifier ) )
230         {
231             if ( !uberzipClassifier.startsWith( "-" ) )
232             {
233                 buf.append( '-' );
234             }
235             buf.append( uberzipClassifier );
236         }
237         buf.append( ".zip" );
238         return buf.toString();
239     }
